import { useState, useEffect } from 'react'

function useDebounce<T>(
  value: T,
  delay: number
): T {
  const [debouncedValue, set] =
    useState<T>(value)

  useEffect(() => {
    const t = setTimeout(() => set(value), delay)
    return () => clearTimeout(t)
  }, [value, delay])
